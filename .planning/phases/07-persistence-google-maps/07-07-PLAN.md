### Create tests/unit/loadEnvKey.test.ts

Test the loadEnvKey function using tmp directories and real files (no mocking needed — the function uses fs directly and is pure):
```ts
import { describe, it, expect, beforeEach, afterEach } from 'vitest';
import { mkdtempSync, writeFileSync, rmSync } from 'node:fs';
import { tmpdir } from 'node:os';
import { join } from 'node:path';
import { loadEnvKey } from '../../src/main/load-env-key.js';

describe('loadEnvKey', () => {
  let dir: string;
  beforeEach(() => { dir = mkdtempSync(join(tmpdir(), 'envtest-')); });
  afterEach(() => { rmSync(dir, { recursive: true, force: true }); });

  it('returns null when .env.local does not exist', () => {
    expect(loadEnvKey(dir)).toBeNull();
  });

  it('parses GOOGLE_MAPS_API_KEY from .env.local', () => {
    writeFileSync(join(dir, '.env.local'), 'GOOGLE_MAPS_API_KEY=AIzaTestKey123\n');
    expect(loadEnvKey(dir)).toBe('AIzaTestKey123');
  });

  it('strips surrounding double quotes from value', () => {
    writeFileSync(join(dir, '.env.local'), 'GOOGLE_MAPS_API_KEY="AIzaQuoted"\n');
    expect(loadEnvKey(dir)).toBe('AIzaQuoted');
  });

  it('strips surrounding single quotes from value', () => {
    writeFileSync(join(dir, '.env.local'), "GOOGLE_MAPS_API_KEY='AIzaSingle'\n");
    expect(loadEnvKey(dir)).toBe('AIzaSingle');
  });

  it('returns null when key is not present in file', () => {
    writeFileSync(join(dir, '.env.local'), 'SOME_OTHER_VAR=foo\n');
    expect(loadEnvKey(dir)).toBeNull();
  });

  it('ignores comment lines', () => {
    writeFileSync(join(dir, '.env.local'), '# GOOGLE_MAPS_API_KEY=commented\nOTHER=x\n');
    expect(loadEnvKey(dir)).toBeNull();
  });

  it('handles = sign in value', () => {
    writeFileSync(join(dir, '.env.local'), 'GOOGLE_MAPS_API_KEY=AIza==base64==\n');
    expect(loadEnvKey(dir)).toBe('AIza==base64==');
  });
});
```

### Create tests/renderer/Favorites.test.tsx

Using happy-dom + @testing-library/react, test the Favorites component:
```tsx
import { render, screen, fireEvent } from '@testing-library/react';
import { describe, it, expect, vi } from 'vitest';
import Favorites, { type Favorite } from '../../src/renderer/src/Favorites';

const sample: Favorite[] = [
  { id: 'a1', name: 'Tokyo', latitude: 35.6762, longitude: 139.6503, createdAt: 1000 },
  { id: 'b2', name: 'London', latitude: 51.5074, longitude: -0.1278, createdAt: 2000 },
];

describe('Favorites', () => {
  it('renders empty state when no favorites', () => {
    render(<Favorites favorites={[]} onSelect={vi.fn()} onDelete={vi.fn()} />);
    expect(screen.getByText(/no favorites/i)).toBeTruthy();
  });

  it('renders favorite rows', () => {
    render(<Favorites favorites={sample} onSelect={vi.fn()} onDelete={vi.fn()} />);
    expect(screen.getAllByTestId('favorite-row')).toHaveLength(2);
    expect(screen.getByText(/Tokyo/)).toBeTruthy();
    expect(screen.getByText(/London/)).toBeTruthy();
  });

  it('calls onSelect with coords on row click', () => {
    const onSelect = vi.fn();
    render(<Favorites favorites={sample} onSelect={onSelect} onDelete={vi.fn()} />);
    fireEvent.click(screen.getAllByTestId('favorite-row')[0]);
    expect(onSelect).toHaveBeenCalledWith({ latitude: 35.6762, longitude: 139.6503 });
  });

  it('calls onDelete with id on delete button click', () => {
    const onDelete = vi.fn();
    render(<Favorites favorites={sample} onSelect={vi.fn()} onDelete={onDelete} />);
    fireEvent.click(screen.getAllByTestId('favorite-delete')[0]);
    expect(onDelete).toHaveBeenCalledWith('a1');
  });

  it('does not call onSelect when delete button is clicked', () => {
    const onSelect = vi.fn();
    const onDelete = vi.fn();
    render(<Favorites favorites={sample} onSelect={onSelect} onDelete={onDelete} />);
    fireEvent.click(screen.getAllByTestId('favorite-delete')[0]);
    expect(onSelect).not.toHaveBeenCalled();
  });
});
```
