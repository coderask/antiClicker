    6. Add handleSetGeo callback:
       const handleSetGeo = async (id: InstanceId, newCoords: PinCoords) => {
         // Optimistic update
         setInstances(prev => {
           const next = new Map(prev);
           const inst = next.get(id);
           if (inst) next.set(id, { ...inst, coords: newCoords });
           return next;
         });
         try {
           await window.api.setGeo(id, newCoords);
         } catch (err) {
           // Revert on failure (MapView will receive old coords via state)
           setInstances(prev => {
             const next = new Map(prev);
             // Can't restore old value here — MapView handles visual revert
             return next;
           });
         }
       };

    7. Add handleClose callback:
       const handleClose = async (id: InstanceId) => {
         try {
           await window.api.close(id);
           // onInstanceClosed event will fire and clean up state
         } catch (err) {
           // If IPC fails, still remove from state
           setInstances(prev => { const n = new Map(prev); n.delete(id); return n; });
           setLiveCount(prev => Math.max(0, prev - 1));
           if (activeId === id) setActiveId(null);
         }
       };

    8. Add handleFocusInstance callback:
       const handleFocusInstance = (id: InstanceId) => {
         setActiveId(id);
         const inst = instances.get(id);
         if (inst) {
           setFlyToTrigger(prev => ({
             ...inst.coords,
             counter: (prev?.counter ?? 0) + 1,
           }));
         }
       };

    9. Add handleRecentPinClick:
       const handleRecentPinClick = (coords: PinCoords) => {
         setDraftPin(coords);
         setFlyToTrigger(prev => ({ ...coords, counter: (prev?.counter ?? 0) + 1 }));
       };
