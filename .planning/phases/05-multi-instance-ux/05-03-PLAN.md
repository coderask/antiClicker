    Implementation changes:

    1. Add `instanceMarkersRef = useRef<Map<string, maplibregl.Marker>>(new Map())`

    2. Add refs for callbacks:
       ```ts
       const onMarkerDragRef = useRef(onMarkerDrag);
       ```

    3. New useEffect for instance markers — deps: [instances, activeId]
       ```ts
       useEffect(() => {
         const map = mapRef.current;
         if (!map) return;

         const incoming = instances ?? new Map();
         const markerMap = instanceMarkersRef.current;

         // Remove markers for instances no longer present
         for (const [id, marker] of markerMap) {
           if (!incoming.has(id)) {
             marker.remove();
             markerMap.delete(id);
           }
         }

         // Add/update markers for current instances
         for (const [id, info] of incoming) {
           if (markerMap.has(id)) {
             // Update position
             markerMap.get(id)!.setLngLat([info.coords.longitude, info.coords.latitude]);
             // Update element style for active state change
             const el = markerMap.get(id)!.getElement();
             const isActive = activeId === id;
             el.style.border = `3px solid ${isActive ? '#ffffff' : 'rgba(255,255,255,0.4)'}`;
             el.style.transform = isActive ? 'scale(1.3)' : 'scale(1)';
           } else {
             // Create new marker
             const el = document.createElement('div');
             const isActive = activeId === id;
             el.style.cssText = [
               'width:20px', 'height:20px', 'border-radius:50%',
               `background:${info.color}`,
               `border:3px solid ${isActive ? '#ffffff' : 'rgba(255,255,255,0.4)'}`,
               'cursor:grab', 'box-shadow:0 2px 6px rgba(0,0,0,0.6)',
               `transform:${isActive ? 'scale(1.3)' : 'scale(1)'}`,
               'transition:transform 0.15s, border-color 0.15s',
             ].join(';');

             const capturedId = id;
             const marker = new maplibregl.Marker({ element: el, draggable: true })
               .setLngLat([info.coords.longitude, info.coords.latitude])
               .addTo(map);

             marker.on('dragend', () => {
               const pos = marker.getLngLat();
               onMarkerDragRef.current?.(capturedId, { latitude: pos.lat, longitude: pos.lng });
             });

             markerMap.set(id, marker);
           }
         }
       }, [instances, activeId]);
       ```
